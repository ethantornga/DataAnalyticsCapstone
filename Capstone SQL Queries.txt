## Salary Difference
SELECT 
    *, 
    expected_post-mba_salary - annual_salary_(before_mba) AS salary_difference
FROM mba_decision_new;

ALTER TABLE mba_decision_new ADD COLUMN salary_difference REAL;

UPDATE mba_decision_new
SET salary_difference = expected_post-mba_salary - annual_salary_(before_mba);



## % of Life Working
SELECT 
    *, 
    years_of_work_experience / age AS percent_of_life_working
FROM mba_decision_new;

ALTER TABLE mba_decision_new ADD COLUMN percent_of_life_working REAL;

UPDATE mba_decision_new
SET percent_of_life_working = years_of_work_experience / age;
